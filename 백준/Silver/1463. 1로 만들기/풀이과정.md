def make_one (X) :
    if X == 1 :
        return 0

    min_operations = float('inf')

    if (X % 3) == 0 :
        min_operations = min(min_operations, make_one(X // 3))
    if (X % 2) == 0 :
        min_operations = min(min_operations, make_one(X // 2))
    
    min_operations = min(min_operations, make_one(X -1))

    return min_operations + 1

X = int(input())
print(make_one (X))
